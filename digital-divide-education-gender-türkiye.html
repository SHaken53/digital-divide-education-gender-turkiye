# Fonksiyon: CSV dosyasini alir, frekans hesaplamalarini yapar, temiz tablo dondurur
hesapla_frekans <- function(dosya_adi) {
  # 1. CSV'yi oku (tum sutunlari karakter olarak al)
  veri <- read_delim(dosya_adi,
                     delim = ";",
                     col_types = cols(.default = col_character()))
  
  # 2. Sutun isimlerini duzenle
  colnames(veri) <- c("Yil", "Toplam", "Erkek", "Kadin")
  
  # 3. Sayisal donusum (virgulu noktaya cevir, karakterden numerige)
  veri <- veri %>%
    mutate(across(c(Toplam, Erkek, Kadin), ~ as.numeric(str_replace(., ",", "."))))
  
  # 4. Sabitler
  group_n <- 120000   # Her egitim duzeyinde toplam birey sayisi
  gender_n <- 60000   # Erkek ve kadin sayisi esit: 60.000
  
  # 5. Hesaplamalar
  veri_frekans <- veri %>%
    mutate(
      Toplam_Kullanan = round(Toplam * group_n / 100),
      Toplam_Kullanmayan = group_n - Toplam_Kullanan,
      
      oran_toplami = Erkek + Kadin,
      Erkek_Pay = Erkek / oran_toplami,
      Kadin_Pay = Kadin / oran_toplami,
      
      Erkek_Kullanan = round(Toplam_Kullanan * Erkek_Pay),
      Kadin_Kullanan = Toplam_Kullanan - Erkek_Kullanan,
      
      Erkek_Kullanmayan = gender_n - Erkek_Kullanan,
      Kadin_Kullanmayan = gender_n - Kadin_Kullanan
    ) %>%
    select(
      Yil,
      Toplam,
      Erkek,
      Kadin,
      Toplam_Kullanan,
      Toplam_Kullanmayan,
      Erkek_Kullanan,
      Erkek_Kullanmayan,
      Kadin_Kullanan,
      Kadin_Kullanmayan
    )
  
  return(veri_frekans)
}

kadin_toplam <- uni_data %>% filter(Cinsiyet == "Kadin") %>% summarise(sum(Frekans)) %>% pull()
if (length(kadin_toplam) == 0) kadin_toplam <- 0

# 3. Oran testi (Z-Test via prop.test)
z_test_sonuc <- prop.test(
